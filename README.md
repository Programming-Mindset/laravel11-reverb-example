# Clone Project 
<pre>
    git clone github.com/Programming-Mindset/laravel11-reverb-example
</pre>
# Install Dependencies
<pre>
    cd laravel11-reverb-example
    npm install
    npm run build

</pre>
# Run Project
<pre>
    php artisan serve
</pre>
